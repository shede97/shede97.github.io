<head>

    <meta charset="UTF-8">

    <title>World Happiness 2024 Narrative Visualization</title>

    <script src="https://d3js.org/d3.v7.min.js"></script>

    <script src="https://unpkg.com/d3-annotation@2"></script>

    <style>

        body { font-family: Arial, sans-serif; text-align: center; }

        .container { margin: 20px auto; max-width: 800px; }

        .controls { margin-bottom: 20px; }

        .button { padding: 10px 20px; margin: 5px; cursor: pointer; background-color: #007bff; color: white; border: none; border-radius: 5px; }

        .button:hover { background-color: #0056b3; }

        .tooltip { position: absolute; background-color: white; border: 1px solid black; padding: 5px; pointer-events: none; }

        .annotation-note-title { font-weight: bold; }

        .annotation-note-label { font-size: 12px; }

    </style>

</head>

<body>

    <div class="container">

        <h1>Exploring 2024 World Happiness Factors</h1>

        <div class="controls">

            <button class="button" onclick="updateScene(1)">Scene 1: Social Support</button>

            <button class="button" onclick="updateScene(2)">Scene 2: Life Expectancy</button>

            <button class="button" onclick="updateScene(3)">Scene 3: Freedom</button>

        </div>

        <svg id="visualization" width="800" height="500"></svg>

    </div>

    <script>

        // Parameters

        let currentScene = 1;

        const scenes = {

            1: {

                x: "Social support",

                y: "Ladder score",

                title: "Happiness vs Social Support",

                annotations: [

                    { note: { title: "High Social Support", label: "Finland: Strong community ties boost happiness" }, x: 1.572, y: 7.741, dy: -30, dx: 50, country: "Finland" },

                    { note: { title: "Low Social Support", label: "Afghanistan: Weak support lowers happiness" }, x: 0.000, y: 1.721, dy: 30, dx: -50, country: "Afghanistan" }

                ]

            },

            2: {

                x: "Healthy life expectancy",

                y: "Ladder score",

                title: "Happiness vs Healthy Life Expectancy",

                annotations: [

                    { note: { title: "High Life Expectancy", label: "Hong Kong: Long, healthy lives correlate with happiness" }, x: 0.857, y: 5.316, dy: -30, dx: 50, country: "Hong Kong S.A.R. of China" }

                ]

            },

            3: {

                x: "Freedom to make life choices",

                y: "Ladder score",

                title: "Happiness vs Freedom",

                annotations: [],

                interactive: true

            }

        };


        // SVG setup

        const svg = d3.select("#visualization");

        const margin = { top: 50, right: 50, bottom: 70, left: 70 };

        const width = svg.attr("width") - margin.left - margin.right;

        const height = svg.attr("height") - margin.top - margin.bottom;

        const g = svg.append("g").attr("transform", `translate(${margin.left},${margin.top})`);


        // Scales

        const xScale = d3.scaleLinear().range([0, width]);

        const yScale = d3.scaleLinear().range([height, 0]);

        const colorScale = d3.scaleOrdinal(d3.schemeCategory10);


        // Axes

        const xAxis = g.append("g").attr("transform", `translate(0,${height})`);

        const yAxis = g.append("g");


        // Tooltip

        const tooltip = d3.select("body").append("div").attr("class", "tooltip").style("opacity", 0);


        // Load data and initialize

        d3.csv("world-happiness-report-2024.csv").then(data => {

            // Convert strings to numbers and filter out invalid rows

            data = data.filter(d => 

                d["Ladder score"] && 

                d["Social support"] && 

                d["Healthy life expectancy"] && 

                d["Freedom to make life choices"] &&

                !isNaN(+d["Ladder score"]) &&

                !isNaN(+d["Social support"]) &&

                !isNaN(+d["Healthy life expectancy"]) &&

                !isNaN(+d["Freedom to make life choices"])

            ).map(d => ({

                "Country name": d["Country name"],

                "Ladder score": +d["Ladder score"],

                "Social support": +d["Social support"],

                "Healthy life expectancy": +d["Healthy life expectancy"],

                "Freedom to make life choices": +d["Freedom to make life choices"]

            }));


            console.log("Processed data:", data); // Debug: Log the processed data


            // Verify annotation countries exist

            const annotationCountries = ["Finland", "Afghanistan", "Hong Kong S.A.R. of China"];

            annotationCountries.forEach(country => {

                if (!data.find(d => d["Country name"] === country)) {

                    console.warn(`Country "${country}" not found in dataset`);

                }

            });


            // Update scene function

            function updateScene(sceneNumber) {

                currentScene = sceneNumber;

                const scene = scenes[sceneNumber];


                // Clear previous content

                g.selectAll(".dot, .annotation").remove();


                // Update scales

                xScale.domain(d3.extent(data, d => d[scene.x])).nice();

                yScale.domain(d3.extent(data, d => d[scene.y])).nice();


                // Update axes

                xAxis.transition().call(d3.axisBottom(xScale));

                yAxis.transition().call(d3.axisLeft(yScale));

                xAxis.append("text")

                    .attr("x", width / 2)

                    .attr("y", 40)

                    .attr("fill", "black")

                    .text(scene.x);

                yAxis.append("text")

                    .attr("x", -height / 2)

                    .attr("y", -40)

                    .attr("transform", "rotate(-90)")

                    .attr("fill", "black")

                    .text(scene.y);


                // Title

                svg.selectAll(".title").remove();

                svg.append("text")

                    .attr("class", "title")

                    .attr("x", width / 2 + margin.left)

                    .attr("y", margin.top / 2)

                    .attr("text-anchor", "middle")

                    .text(scene.title);


                // Draw dots

                g.selectAll(".dot")

                    .data(data)

                    .enter()

                    .append("circle")

                    .attr("class", "dot")

                    .attr("cx", d => xScale(d[scene.x]))

                    .attr("cy", d => yScale(d[scene.y]))

                    .attr("r", 5)

                    .attr("fill", (d, i) => colorScale(i % 10))

                    .on("mouseover", function(event, d) {

                        if (scene.interactive) {

                            tooltip.transition().style("opacity", 0.9);

                            tooltip.html(`Country: ${d["Country name"]}<br>Ladder score: ${d["Ladder score"]}<br>${scene.x}: ${d[scene.x]}`)

                                .style("left", (event.pageX + 5) + "px")

                                .style("top", (event.pageY - 28) + "px");

                        }

                    })

                    .on("mouseout", function() {

                        if (scene.interactive) {

                            tooltip.transition().style("opacity", 0);

                        }

                    });


                // Annotations

                const annotations = scene.annotations

                    .map(ann => {

                        const countryData = data.find(d => d["Country name"] === ann.country);

                        if (!countryData) {

                            console.warn(`Annotation for ${ann.country} skipped: country not found or missing data`);

                            return null;

                        }

                        return {

                            ...ann,

                            x: countryData[scene.x],

                            y: countryData[scene.y]

                        };

                    })

                    .filter(ann => ann !== null);

                const makeAnnotations = d3.annotation()

                    .type(d3.annotationLabel)

                    .accessors({

                        x: d => xScale(d.x),

                        y: d => yScale(d.y)

                    })

                    .annotations(annotations);

                g.append("g")

                    .attr("class", "annotation")

                    .call(makeAnnotations);

            }


            // Initial scene

            updateScene(1);

        }).catch(error => {

            console.error("Error loading CSV:", error);

        });

    </script>

</body>

</html>
